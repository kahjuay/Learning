Function AlignCountryRow(strSearchString As String, strLocStart As String, strLocEnd As String) As Integer
Dim intLastRow As Integer
On Error GoTo errHandler
    Range(strLocStart & ":" & strLocEnd).Select
    Selection.Find(What:=strSearchString, After:=ActiveCell, LookIn:=xlFormulas, _
        LookAt:=xlPart, SearchOrder:=xlByRows, SearchDirection:=xlNext, _
        MatchCase:=False, SearchFormat:=False).Activate
    Selection.FindNext(After:=ActiveCell).Activate

    AlignCountryRow = ActiveCell.Row
    
errHandler:
    If Err.Number = 91 Then
    'Can't find country in main country row, add to it
        intLastRow = FindLastRow(1, intLastRowConst)
        Cells(intLastRow + 1, 1).Value = strSearchString
        AlignCountryRow = intLastRow + 1
    End If
End Function

Sub AdjustMVCTSD(strFund As String, strPTFile As String)
Dim intFirstRow As Integer
Dim intLastRow As Integer
Dim intCounter As Integer
Dim strData As String
Dim strWSName As String
Dim strCountry As String
Dim strSector As String
Dim intSourceCol As Integer
Dim intTargetCol As Integer
Dim arrData As Variant
Dim strKey As String
Dim dblMVAdj As Double
Dim dblCTSDAdj As Double

    'Financial Adjustments from HK into China
    'Go through HK names in Detail Report, pick up ticker (col C) of Finance Sector (col P) securities
    'Go to DurbuckCTDOV_Index Exposure Portfolio section and pick up MV (col H) and CTSD (col J) adjustments
    strData = ""
    strWSName = "Detail2"
    strCountry = "Hong Kong"
    strSector = "FINANCIAL"
    intSourceCol = 16
    intTargetCol = 3
    
    Workbooks(strPTFile).Worksheets(strWSName).Activate
    
    intFirstRow = FindCountryRow(strCountry)
    intLastRow = FindCountryRow(strCountry & " Subtotal")
    
    For intCounter = intFirstRow + 1 To intLastRow - 1
        If Cells(intCounter, 16).Value = strSector Then
            strData = strData & "," & Cells(intCounter, 3).Value
        End If
    Next intCounter

    arrData = Split(strData, ",")
    
    strWSName = "DurbuckCTDOV_Index Exposure"
    Workbooks(strPTFile).Worksheets(strWSName).Activate
    
    strKey = "Portfolio"
    intFirstRow = FindCountryRow(strKey)
    intLastRow = FindLastRow(2, intLastRowConst)
    dblMVAdj = 0
    dblCTSDAdj = 0
    
    For intCounter = intFirstRow + 1 To intLastRow - 1
        If IsInArray(Cells(intCounter, 2).Value, arrData) > 0 Then
            'Found in array
            dblMVAdj = dblMVAdj + Cells(intCounter, 8).Value
            dblCTSDAdj = dblCTSDAdj + Cells(intCounter, 10).Value
        End If
    Next intCounter
    
    'strData = dblMVAdj & "," & dblCTSDAdj
    
    'arrData = Split(strData, ",")
    
    Workbooks(strComboWBName).Worksheets(strComboWSName).Activate
    
    strKey = "China Active"
    intFirstRow = FindCountryRow(strKey)
    intLastRow = FindLastRow(intMiddleCol, intLastRowConst)
    
    For intCounter = intFirstRow + 1 To intLastRow
        If Cells(intCounter, intMiddleCol).Value = strSector Then
            If strFund = strFund1 Then
                Cells(intCounter, intMiddleCol + 1).Formula = "=" & Cells(intCounter, intMiddleCol + 1).Value & "+" & dblMVAdj
                Cells(intCounter, intMiddleCol + 3).Formula = "=" & Cells(intCounter, intMiddleCol + 3).Value & "+" & dblCTSDAdj
            Else
                Cells(intCounter, intMiddleCol + 2).Formula = "=" & Cells(intCounter, intMiddleCol + 2).Value & "+" & dblMVAdj
                Cells(intCounter, intMiddleCol + 4).Formula = "=" & Cells(intCounter, intMiddleCol + 4).Value & "+" & dblCTSDAdj
            End If
        End If
    Next intCounter
    
    'Quasi and Oil Gas Adjustments
    'Go to Sector Report, pick up MV Exposure in Col W, CTSD Exposure at AI
    'Add to Oil Gas Sector and reduce from Quasi Sov sector
    
    strWSName = "Sector"
    Workbooks(strPTFile).Worksheets(strWSName).Activate
    strKey = "Manual adjustments for China from Quasi Sov into Oil and Gas"
    
    intFirstRow = FindCountryRow(strKey)
    
    dblMVAdj = Cells(intFirstRow + 3, 23).Value
    dblCTSDAdj = Cells(intFirstRow + 3, 35).Value
    
    Workbooks(strComboWBName).Worksheets(strComboWSName).Activate
    
    strKey = "China Active"
    intFirstRow = FindCountryRow(strKey)
    intLastRow = FindLastRow(intMiddleCol, intLastRowConst)
    
    For intCounter = intFirstRow + 1 To intLastRow
        If Cells(intCounter, intMiddleCol).Value = "OIL_GAS" Then
            If strFund = strFund1 Then
                Cells(intCounter, intMiddleCol + 1).Formula = "=" & Cells(intCounter, intMiddleCol + 1).Value & "+" & dblMVAdj
                Cells(intCounter, intMiddleCol + 3).Formula = "=" & Cells(intCounter, intMiddleCol + 3).Value & "+" & dblCTSDAdj
            Else
                Cells(intCounter, intMiddleCol + 2).Formula = "=" & Cells(intCounter, intMiddleCol + 2).Value & "+" & dblMVAdj
                Cells(intCounter, intMiddleCol + 4).Formula = "=" & Cells(intCounter, intMiddleCol + 4).Value & "+" & dblCTSDAdj
            End If
        ElseIf Cells(intCounter, intMiddleCol).Value = "QUASI_SOV" Then
            If strFund = strFund1 Then
                Cells(intCounter, intMiddleCol + 1).Formula = "=" & Cells(intCounter, intMiddleCol + 1).Value & "-" & dblMVAdj
                Cells(intCounter, intMiddleCol + 3).Formula = "=" & Cells(intCounter, intMiddleCol + 3).Value & "-" & dblCTSDAdj
            Else
                Cells(intCounter, intMiddleCol + 2).Formula = "=" & Cells(intCounter, intMiddleCol + 2).Value & "-" & dblMVAdj
                Cells(intCounter, intMiddleCol + 4).Formula = "=" & Cells(intCounter, intMiddleCol + 4).Value & "-" & dblCTSDAdj
            End If
        End If
    Next intCounter
    
    
End Sub



Function IsInArray(stringToBeFound As String, arr As Variant) As Long
  Dim i As Long
  ' default return value if value not found in array
  IsInArray = -1

  For i = LBound(arr) To UBound(arr)
    If StrComp(stringToBeFound, arr(i), vbTextCompare) = 0 Then
      IsInArray = i
      Exit For
    End If
  Next i
End Function

Function SumCash(strPTFile As String, intCol As Integer) As Variant
Dim dblCashTotal As Double
Dim dblCashMVTotal As Double
Dim intLastRow As Integer
Dim intRowCounter As Integer
Dim varCash As Variant
Dim strWSSource As String
Dim intUSTCol As Integer
Dim intParCol As Integer
Dim intMVCol As Integer

strWSSource = "Detail2"

Workbooks(strPTFile).Worksheets(strWSSource).Activate
intLastRow = FindLastRow(intCol, intLastRowConst)

dblCashTotal = 0
dblCashMVTotal = 0
intUSTCol = 4
intParCol = 25
intMVCol = 19
For intRowCounter = 1 To intLastRow
    If ((InStr(1, Cells(intRowCounter, 2).Value, "CASHUSD") > 0) Or (InStr(1, Cells(intRowCounter, intUSTCol).Value, "TREASURY BILL") > 0) Or (InStr(1, Cells(intRowCounter, intUSTCol).Value, "US TREASURY") > 0)) Then
        'If Cash is found then add to bucket

        dblCashMVTotal = dblCashMVTotal + Cells(intRowCounter, intMVCol).Value
    End If
Next intRowCounter

varCash = Array(dblCashTotal, dblCashMVTotal)

SumCash = varCash
End Function


Function VerifyColumns(strWkSheet As String) As Integer
Dim intRecordRow As Integer
Dim intRecordCol As Integer
Dim intStartRow As Integer
Dim intStartCol As Integer
Dim intEndCol As Integer
Dim varSheetVerify As Variant
Dim varSheetTarget As Variant
Dim strVerifyRange As String
Dim strRangeToCheck As String
Dim iRow As Integer
Dim iCol As Integer

    Workbooks(strMacroFileName).Worksheets("Verification").Activate
    
    'Find Verification Record
    Cells.Find(What:=strWkSheet, After:=ActiveCell, LookIn:=xlFormulas, _
        LookAt:=xlPart, SearchOrder:=xlByRows, SearchDirection:=xlNext, _
        MatchCase:=False, SearchFormat:=False).Activate
    
    intRecordRow = ActiveCell.Row
    'MsgBox intRecordRow
    
    ActiveCell.Offset(0, 1).Select
    intStartRow = ActiveCell.Value
    ActiveCell.Offset(0, 1).Select
    intStartCol = ActiveCell.Value
    ActiveCell.Offset(0, 1).Select
    intEndCol = ActiveCell.Value
    ActiveCell.Offset(0, 1).Select
    intRecordCol = ActiveCell.Column
    
    strVerifyRange = Cells(intRecordRow, intRecordCol).Address & ":" & Cells(intRecordRow, intRecordCol + intEndCol).Address
    strRangeToCheck = Cells(intStartRow, intStartCol).Address & ":" & Cells(intStartRow, intEndCol).Address
    'MsgBox strRangeToCheck
    
    
    varSheetVerify = ActiveSheet.Range(strVerifyRange)
    
    Workbooks(strPTFileName).Worksheets(strWkSheet).Activate
    
    varSheetTarget = ActiveSheet.Range(strRangeToCheck)
    
    VerifyColumns = 0
    
    For iRow = LBound(varSheetVerify, 1) To UBound(varSheetVerify, 1)
        'For iCol = LBound(varSheetVerify, 2) To UBound(varSheetVerify, 2)
        For iCol = 1 To intEndCol - intStartCol + 1
            If varSheetVerify(iRow, iCol) = varSheetTarget(iRow, iCol) Then
                ' Cells are identical.
                ' Do nothing.
                VerifyColumns = VerifyColumns + 0
            Else
                'Some discrepancy detected with the PT file vs verify page
                VerifyColumns = VerifyColumns + 1
            End If
        Next
    Next
    
'    If VerifyColumns > 0 Then
'        MsgBox "There are discrepancies found within the column headers of " & strWkSheet
'    End If
    
End Function

Sub ProcessJaci()
Dim strWSName As String
Dim intLastRow As Integer
Dim WB As Workbook
Dim WS As Worksheet
Dim strJACI As String


On Error GoTo ErrHandle


Set WB = ActiveWorkbook

    For Each WS In WB.Worksheets
        strJACI = WS.Name
        'MsgBox strJACI
        If Left(strJACI, 4) = "jaci" Then
            WS.Name = "jaci"
            'strWSName = strJACI
        End If
    Next WS


    strWSName = "jaci"
    'Check if Jaci worksheet exists, exit loop if not found
    Worksheets(strWSName).Activate
    
MainProcess:
    Worksheets(strWSName).Activate
    
    If VerifyColumns(Left(strWSName, 4)) > 0 Then
        MsgBox "Hi " & strUserName & ", I found column discrepancies in " & strWSName & " and will skip and move on to the next available worksheet"
        Exit Sub
    End If
    
    'Check whether processing was previously done
    If Cells(1, 22).Value = "CTSD" Then
        MsgBox "Hi, " & strUserName & ", " & strWSName & " was previously processed, skipping to the next available worksheet"
        Exit Sub
    End If
    

    intLastRow = FindLastRow(1, intLastRowConst)
    
    'Add CTSD, PX_BID column and cookie column
    Columns("V:V").Select
    'CTSD column in V
    Selection.Insert Shift:=xlToRight, CopyOrigin:=xlFormatFromLeftOrAbove
    Range("V1").Select
    ActiveCell.FormulaR1C1 = "CTSD"
    Range("V2").Select
    ActiveCell.FormulaR1C1 = "=RC[-6]*RC[-1]/100"
    Range("V2").Select
    'Selection.AutoFill Destination:=Range("V2:V940")
    Selection.AutoFill Destination:=Range("V2:V" & intLastRow)
    'Range("V1:V940").Select
    Columns("V:V").Select
    Selection.ColumnWidth = 24.6
    With Selection.Font
        'Make this column red
        .Color = -16776961
        .TintAndShade = 0
    End With
    
    Columns("AA:AA").Select
    'px_bid column in AA
    Selection.Insert Shift:=xlToRight, CopyOrigin:=xlFormatFromLeftOrAbove
    Range("AA1").Select
    ActiveCell.FormulaR1C1 = "px_bid"
    
    Range("AA2").Select
    ActiveCell.FormulaR1C1 = "=BDP(RC[-24]&"" CORP"", ""PX_BID"")"
    
    Columns("AB:AB").Select
    'cookie column in AB
    Selection.Insert Shift:=xlToRight, CopyOrigin:=xlFormatFromLeftOrAbove
    Range("AB1").Select
    ActiveCell.FormulaR1C1 = "cookie"
    
    Range("AB2").Select
    ActiveCell.FormulaR1C1 = "=(RC[-1]/RC[-2]-1)*100*RC[-12]"
    
    
    Range("AA2:AB2").Select
    'Selection.AutoFill Destination:=Range("AA2:AB940")
    Selection.AutoFill Destination:=Range("AA2:AB" & intLastRow)
    
    Columns("AA:AB").Select
    Selection.ColumnWidth = 24.6
    With Selection.Font
        'Make this column red
        .Color = -16776961
        .TintAndShade = 0
    End With
      
     
    
    With Worksheets(strWSName).Tab
        .Color = 12611584 'Blue
        '.Color = xlAutomatic
        .TintAndShade = 0
    End With
    
    Rows("2:2").Select
    Selection.Delete Shift:=xlUp
    
    strProcessedSheets = strProcessedSheets + vbCrLf + strWSName
    Exit Sub
ErrHandle:
    'MsgBox Err.Number & " " & Err.Description
    If Err.Number = 9 Then
        'jaci worksheet not found
        'MsgBox strWSName & " worksheet not found, skipping to next available worksheet"
        GoTo CheckJaciWB
    End If
    
CheckJaciWB:
    'Checking whether jaci workbook is available in the background
    'Call SwitchWB("jaci")
    Call SwitchWB("JPM_JACI")
    If intGoodFile = 1 Then
        Exit Sub
    Else
    'Switch to jaci workbook, copy it into PT workbook and rename tab to jaci
    
        'strJACI = Left(strJaciFileName, InStr(strJaciFileName, ".") - 1)
        strJACI = ActiveSheet.Name

        Sheets(strJACI).Select
        Sheets(strJACI).Copy Before:=Workbooks( _
        strPTFileName).Sheets(1)
        Sheets(strJACI).Select
        Sheets(strJACI).Name = "jaci"
    
        Resume MainProcess
    End If
End Sub

Sub UpdateAdjCTDOVIdxExp(strFundCode As String, strIndexPort As String)
Dim intFirstRow As Integer
Dim intLastRow As Integer
Dim intCounter As Integer
Dim strWSSource As String
Dim intBucketCol As Integer
Dim intTickerCol As Integer
Dim intISINCol As Integer
Dim intCTSDCol As Integer
Dim intTickerCount As Integer
Dim strTickerList As String
Dim strTickerSelected As String
Dim arrData As Variant
Dim dbl2y As Double
Dim dbl5y As Double
Dim dbl10y As Double
Dim dbl30y As Double
Dim strISINList As String
Dim arrISIN As Variant
Dim intISINCount As Integer
Dim varISINTest As Variant
Dim bolISINFound As Boolean

    'Switch back to Macro Workbook
    Application.Workbooks(strMacroFileName).Activate
    'Update Adj_CTDOVIdxExp Index values
    strWSSource = "Adj_CTDOVIdxExp"
    Worksheets(strWSSource).Activate
    
    intFirstRow = FindCountryRow(strFundCode & " start")
    intLastRow = FindCountryRow(strFundCode & " end")
    
    'Determining which column stores the tickers depending on whether we are looking at Index or Portfolio
    'This are the positions within Macro file
    If strIndexPort = "Index" Then
        intTickerCol = 1
    Else
        'Updating portfolios
        intTickerCol = 8
    End If
    
    
    
    strTickerList = ""
    For intCounter = intFirstRow + 2 To intLastRow - 2
        'Collect Tickers
        If Cells(intCounter, intTickerCol).Value <> "" And Cells(intCounter, intTickerCol).Value <> "Total" Then
            strTickerList = Cells(intCounter, intTickerCol).Value & "," & strTickerList
        End If
    Next intCounter
    
    'Remove last comma
    strTickerList = Left(strTickerList, Len(strTickerList) - 1)

    'Putting the list of tickers within the index as an array
    arrData = Split(strTickerList, ",")
    
    intISINCol = 16
    strISINList = ""
    
    intFirstRow = 3
    intLastRow = FindLastRow(16, intLastRowConst)
    
    For intCounter = intFirstRow To intLastRow
        'Collect ISIN
        strISINList = Cells(intCounter, intISINCol).Value & "," & strISINList
    Next intCounter
    
    strISINList = Left(strISINList, Len(strISINList) - 1)
    arrISIN = Split(strISINList, ",")
    
    'Run through each member of array, generating CTD according to spread duration bucket
    For intTickerCount = LBound(arrData) To UBound(arrData)
    
    
    If strIndexPort = "Index" Then
        'Flip to the PT File, jaci worksheet and highlight securities that matched any member of arrData
        Application.Workbooks(strPTFileName).Activate
        
        strWSSource = "jaci"
        Worksheets(strWSSource).Activate
        'jaci worksheet holds CTSD in column 22
        intCTSDCol = 22
        intBucketCol = 21
        intTickerCol = 4
        intFirstRow = 2
        intISINCol = 3
    Else
        'Updating portfolios
        Application.Workbooks(strPTFileName).Activate
        
        strWSSource = "Detail2"
        Worksheets(strWSSource).Activate
        intCTSDCol = 34
        intBucketCol = 8
        intTickerCol = 3
        intISINCol = 2
        intFirstRow = FindCountryRow("Country Of Risk Name") + 1
    End If
    
    intLastRow = FindLastRow(1, intLastRowConst)
    
    dbl2y = 0
    dbl5y = 0
    dbl10y = 0
    dbl30y = 0
    
        For intCounter = intFirstRow To intLastRow
            strTickerSelected = arrData(intTickerCount)
            If Cells(intCounter, intTickerCol).Value = strTickerSelected Then
            'Matching row of ticker found
            
                'Check whether ISIN list is matched.  Update both Index and Portfolio boxes only if matched
                bolISINFound = False
                varISINTest = Filter(arrISIN, Cells(intCounter, intISINCol).Value)
                
                If UBound(varISINTest) >= 0 And Cells(intCounter, intISINCol).Value <> "" Then
                    'ISIN Matched
                    bolISINFound = True
                End If
            
                If bolISINFound = True Then
                    'Found in array, highlight in green
                    'Call ColorRow(intCounter, "Green")
                    
                    'Put it into correct CTD bucket depending on spread duration value (col 21 for jaci)
                    Select Case Cells(intCounter, intBucketCol).Value
                        Case Is <= 2
                            dbl2y = dbl2y + Cells(intCounter, intCTSDCol).Value
                        Case Is <= 5
                            dbl5y = dbl5y + Cells(intCounter, intCTSDCol).Value
                        Case Is <= 10
                            dbl10y = dbl10y + Cells(intCounter, intCTSDCol).Value
                        Case Else
                            dbl30y = dbl30y + Cells(intCounter, intCTSDCol).Value
                    End Select
                End If
            End If
        Next intCounter
        'CTD of ticker calculated, flip back to Macro worksheet and update the table
        
       
        
        
        Call UpdateAdjCTDOVIdxExpDetails(strFundCode, strIndexPort, strTickerSelected, dbl2y, dbl5y, dbl10y, dbl30y)
    Next intTickerCount
    
    
    


End Sub

                ActiveCell.Formula = "=(W" & intCounter & ") * 1000000 /" & "W" & intTotalRow & "/10 * L" & intCounter
            End If

        .TintAndShade = 0
    End With
    
    strProcessedSheets = strProcessedSheets + vbCrLf + strWSName

Sub TriggerManualIndexPortOverride()
Dim intFirstCountryRow As Integer
Dim intFirstRow As Integer
Dim intLastRow As Integer
Dim strWSName As String
Dim intCounter As Integer
Dim strFormula As String
Dim intCountryCount As Integer

'Coming to this step would mean that the form was previously processed
    strWSName = "DurbuckCTDOV_Index Exposure"
    intFirstCountryRow = FindLastRow(2, intLastRowConst)
    intLastRow = intFirstCountryRow
    
    'Need to do this 5 times to get first country row
    For intCounter = 1 To 5
        intFirstCountryRow = FindFirstRowReversed(2)
        If intCounter = 3 Then
            intFirstRow = intFirstCountryRow
        End If
    Next intCounter
    
    'Remove the old table block
    Range("B" & intLastRow + 1 & ":G" & intLastRow + 1).Copy Range("B" & intFirstRow & ":G" & intLastRow)
    
    'Remove the formula modifications for China and HK
    'Need to clear this 7 times for each country
    For intCountryCount = 1 To 2
        For intCounter = 1 To 7
            Cells(intFirstCountryRow, 2 + intCounter).Select
            If InStr(ActiveCell.Formula, "-") > 0 Then
                ActiveCell.Formula = Left(ActiveCell.Formula, Application.WorksheetFunction.Min(InStr(ActiveCell.Formula, "-") - 1, InStr(ActiveCell.Formula, "+") - 1))
            End If
        Next intCounter
        intFirstCountryRow = intFirstCountryRow + 1
    Next intCountryCount
    
    'Reset intFirstCountryRow
    intFirstCountryRow = FindFirstRowReversed(2) + 1
    Call ManualIndexPortOverride(intFirstCountryRow, strWSName)
End Sub

Function FindCountryRow(strCountry As String)

    Range("A1").Select
    Cells.Find(What:=strCountry, After:=ActiveCell, LookIn:=xlFormulas, LookAt _
        :=xlPart, SearchOrder:=xlByRows, SearchDirection:=xlNext, MatchCase:= _
        False, SearchFormat:=False).Activate
    FindCountryRow = ActiveCell.Row
    Exit Function
errHandler:
If Err.Number = 91 Then
    MsgBox "Oops, sorry " & strUserName & " I can't find " & strCountry
    intGoodFile = 1
End If
End Function

Function HideWorksheet(strWkSheet)
On Error GoTo ErrHandle

    Worksheets(strWkSheet).Activate
    If ActiveSheet.Visible < 0 Then
        Sheets(strWkSheet).Select
        ActiveWindow.SelectedSheets.Visible = xlSheetHidden
    End If

ErrHandle:
If Err.Number = 9 Then
    'Can't find worksheets to hide, so likely looking at wrong report book
    intGoodFile = 1
End If

End Function
